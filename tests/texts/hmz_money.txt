𖼄𖽔𖼷𖽺𖼷𖽷𖾑 𖼮𖽑𖽢𖼄𖽔 𖼄𖽑𖽪𖽾𖾑
𖼊𖽘𖾐𖼊𖽑𖽙𖽪𖾐𖼽𖽝𖾐，𖼄𖽔𖽃𖽡𖾑𖼀𖽡𖽝𖾏𖽃𖽔𖾏𖼊𖽑𖽺 𖼄𖽔𖼷𖽺𖼊𖽝𖾑 𖼄𖽑𖽪𖽾𖾑，𖼄𖽔𖽃𖽡𖾑𖼊𖽑𖽙𖼵𖽪𖽾𖾏𖼽𖽑𖽝𖼖𖽑𖽔，𖼊𖽘𖼐𖽡𖼊𖽝𖾑𖼊𖽑𖽙𖼽𖽑𖽝 𖼖𖽑𖽔𖼐𖽔𖾐𖼷𖽑𖽺𖼄𖽔𖼃𖽙𖽪𖾑𖼮𖽡𖽻𖾐𖼊𖽙𖾑，𖼷𖽷𖾑𖼄𖽔 𖼷𖽺，𖼄𖽔𖽃𖽡𖾑 𖼐𖽑𖽙𖾑，𖼊𖽑𖽙𖼄𖽔𖼷𖽺 𖼄𖽑𖽪𖽾𖾑𖼐𖽔𖾐𖼽𖽙𖼐𖽡𖼊𖽑𖽙𖼽𖽑𖽝𖼖𖽑𖽔 𖼐𖽔𖾐𖼚𖽔𖾏 𖼐𖽡𖼈𖽝𖾐𖼄𖽪𖽃𖽔𖾏𖼺𖽷𖾏．𖼊𖽑𖽙𖼽𖽑𖽝𖼖𖽑𖽔𖼐𖽔𖾐𖼊𖽝𖾐𖼚𖽔𖾏𖼐𖽡𖼈𖽝𖾐𖼄𖽪，𖽃𖽡𖾑 𖼄𖽪𖼷𖽑𖽻，𖼊𖽝𖾐𖼨𖽑𖽙𖾏𖼐𖽻𖼐𖽑𖽻𖾏 ，𖼣𖽡𖽻𖾑𖼐𖽻 𖼐𖽑𖽻𖾐𖼊𖽑𖽙𖽪𖾐𖼽𖽝𖾐，𖼐𖽡𖼀𖽻𖽃𖽡𖾑𖼖𖽙𖾑𖼞𖽔𖼐𖽡𖽐𖼊𖽱𖼷𖽷𖾑𖼷𖽑𖽻𖼊𖽝𖾑𖼐𖽙，𖼊𖽝𖾐𖼽𖽑𖽻𖼐𖽡𖼄𖽔𖼷𖽺𖼊𖽝𖾑𖼽𖽑𖽝𖼖𖽑𖽔 𖼄𖽔𖾑 𖼃𖽙𖽪𖾑 𖼖𖽙𖾑𖼣𖽺𖼖𖽑𖽻𖽐𖼮𖽑𖽺 𖼃𖽙𖽪𖾑 𖼷𖽑𖽺 𖼡𖽔𖾑𖼮𖽑𖽝𖼎𖽪𖽾𖼐𖽻𖼐𖽑𖽻𖾏，𖼐𖽡𖼊𖽝𖾐𖼊𖽙𖾏𖼺𖽺𖾑𖼡𖽑𖽙𖾑：“𖼷𖽷𖾑𖼄𖽔𖼷𖽺𖼊𖽝𖾑 𖼄𖽑𖽪𖽾𖾑 𖼐𖽑𖽙𖾑𖼣𖽪𖽾𖼐𖽘𖼚𖽔𖾏，𖼄𖽔𖼷𖽺𖼊𖽝𖾑 𖼄𖽑𖽪𖽾𖾑𖼵𖽪𖽾𖾏𖼚𖽔𖾏 𖼐𖽑𖽙𖾑 𖼣𖽪𖽾．
       𖼚𖽔𖾏𖼊𖽑𖽘𖽃𖽻𖾑𖼀𖽝𖾑𖼺𖽪𖽾𖾐，𖼊𖽑𖽙𖼷𖽷𖾑𖼄𖽔𖼷𖽺𖼊𖽝𖾑 𖼄𖽑𖽪𖽾𖾑𖼐𖽔𖾐𖼊𖽝𖾐𖼽𖽙𖼊𖽑𖽙𖼄𖽔𖼷𖽺𖼊𖽝𖾑𖼽𖽑𖽝𖼖𖽑𖽔𖼐𖽔𖾐𖼚𖽔𖾏𖼐𖽡 𖼈𖽝𖾐 𖼄𖽪𖽃𖽔𖾏𖼺𖽷𖾏， 𖼊𖽑𖽙𖼽𖽑𖽝𖼖𖽑𖽔𖼐𖽔𖾐𖼊𖽝𖾐 𖼄𖽪𖼖𖽝𖾐．𖽃𖽡𖾑𖼊𖽑𖽙𖽪𖼷𖽷𖾐𖼊𖽝𖾐𖼄𖽔𖾑𖼡𖽔𖾑𖼎𖽪𖽾𖼐𖽑𖽻𖾏𖼊𖽑𖽺 𖼎𖽙𖽪𖾏𖼀𖽡𖽝𖾐𖼊𖽘𖼨𖽑𖽙𖾏𖼐𖽻 𖼐𖽑𖽻𖾏， 𖼣𖽡𖽻𖾑𖼐𖽻𖼐𖽑𖽻𖾏𖼊𖽝𖾑𖼊𖽑𖽙𖽪𖾐𖼽𖽝𖾐，𖼐𖽡𖼨𖽑𖽙𖾏 𖼃𖽙𖽪𖾑𖼮𖽡𖽻𖾐 𖼊𖽙𖾑𖼊𖽔𖼡𖽑𖽙𖾑： “𖼦𖽔𖾐 𖼐𖽔𖾐𖼊𖽑𖽺 𖽐𖼡𖽺，𖼞𖽻𖾑𖼊𖽻𖾐𖼣𖽡𖾏𖼃𖽙𖽪𖾑𖼖𖽑𖽞𖼞𖽱𖾐 𖼊𖽡𖼊𖽑𖽺 𖼃𖽙𖽪𖾑𖼷𖽑𖽺𖼡𖽔𖾑𖼮𖽑𖽝𖼎𖽪𖽾， 𖼊𖽑𖽺𖼚𖽔𖾏𖼡𖽔𖾑𖼃𖽑𖽘𖼄𖽪𖼐𖽻”．𖽃𖽡𖾑𖼊𖽙𖾑𖼄𖽡𖾐𖼐𖽡𖼮𖽷𖾐𖼽𖽨𖾐𖼡𖽑𖽙𖾑：“𖼖𖽙𖾑𖼡𖽔𖾑𖼎𖽪𖽾 𖼐𖽔𖾐𖼡𖽑𖽝𖼦𖽞𖾏，𖼣𖽡𖾏𖼃𖽙𖽪𖾑𖼊𖽑𖽙𖽃𖽡𖾑 𖼊𖽑𖽙 𖼊𖽑𖽺 𖽃𖽡𖾑𖼷𖽑𖽺𖾐𖼊𖽠𖾏，𖼊𖽑𖽺 𖼺𖽠𖾐𖼖𖽑𖽻𖼀𖽙𖾑𖼀𖽝𖾑𖼐𖽻，𖼃𖽙𖽪𖾑𖼖𖽑𖽞𖼞𖽱𖾐𖼊𖽡𖽃𖽡𖾑𖼞𖽝𖾏𖼄𖽔𖾑𖼡𖽔𖾑𖼎𖽪𖽾𖼊𖽑𖽺 𖼎𖽪𖽾𖼊𖽑𖽺 𖼺𖽠𖾐 𖼖𖽑𖽻”． 𖽃𖽔𖾏𖼊𖽑𖽙𖼄𖽔 𖼷𖽺𖼊𖽝𖾑𖼽𖽑𖽝𖼖𖽑𖽔𖼽𖽝𖾐𖼊𖽙𖾏𖼺𖽺𖾑𖼡𖽑𖽙𖾑：“𖽃𖽺𖾏！𖼄𖽑𖽪𖽾𖾑𖼽𖽑𖽻𖼮𖽡𖽺𖼊𖽝𖾑，𖼷𖽺𖼽𖽑𖽻𖼊𖽑𖽔𖼊𖽝𖾑，𖼄𖽔 𖼄𖽑𖽪𖽾𖾑𖼊𖽝𖾐𖼄𖽔𖼷𖽑𖽔𖾏𖼷𖽘𖾐，𖼷𖽷𖾑𖼄𖽔𖼄𖽑𖽪𖽾𖾑，𖽃𖽡𖾑𖼷𖽘𖾏𖼊𖽝𖾐𖼷𖽷𖾑𖼄𖽔”．
 
𖼺𖽙𖾐𖽃𖽡𖾐，𖼄𖽔𖽃𖽡𖾑𖼖𖽻𖾐𖼺𖽵𖼖𖽑𖽙𖼡𖽑𖽙𖾑：‘𖼄𖽔𖼷𖽺𖼷𖽷𖾑𖼺𖽪𖽘𖼄𖽔，𖼷𖽷𖾑𖼄𖽔𖼷𖽺𖼽𖽝𖾐𖼷𖽷𖾑𖼺𖽪𖽘𖼺𖽠𖾐．
